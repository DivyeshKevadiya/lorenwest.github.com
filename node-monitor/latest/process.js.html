var sys = require('sys'),
    events = require('events'),
    Monitor = require('./monitor'),
    Probe = require('./probe');

// Define and register the class
var classDefinition = {
  id: null,
  name: 'Process',
  description: 'Probes all process related information and events.  Static process information (pid, etc) is located in the probe preferences.  Monitor events are generated on a polling interval.',
  defaultParameters: {
    pollInterval: '0 * * * * *'
  },
  parameterDescriptions: {
    pollInterval: 'Polling interval for generating monitor events. The cron pattern consists of six fields separated by spaces. The format is: [second] [minute] [hour] [day of month] [month] [day of week].  More about cron formats here: http://help.sap.com/saphelp_xmii120/helpdata/en/44/89a17188cc6fb5e10000000a155369/content.htm'
  }
};

/**
 * Probe for application process information.
 *
 * @class Process
 * @extends Probe
 * @constructor
 * @param {String} name - A name identifying the instance of this monitor.
 * @param {Object} config - Configuration override object
 */
var Process = module.exports = function (name, config) {
  Probe.call(this, classId, name, config);
}
sys.inherits(Process, Probe);
var classId = Monitor.registerClass(Process, classDefinition);

/**
 * Start the monitor
 *
 * @method start
 * @param {Function} callback(error) - (optional) Called when the monitor has started
 */
Process.prototype.superStart = Process.prototype.start;
Process.prototype.start = function(callback) {
  var t = this;
  t.superStart(callback);
}
