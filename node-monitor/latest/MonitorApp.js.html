YUI.add("monitorapp", function(Y) {

  /**
   * Browser-side elements for the Monitor UI.
   *
   * @module MonitorUI-Client
   */

  /**
   * <p>MonitorApp is a 
   *   <a href="http://developer.yahoo.com/yui/3/api/Widget.html">YUI Widget</a> 
   * class representing the monitor application UI.
   * </p>
   * 
   * <p>The widget is usually placed within the top element of a blank page.</p>
   * 
   * <script type="text/javascript">
   *   document.getElementById("showProtected").style.display = "block";
   * </script>
   *
   * @class MonitorApp
   * @extends YUI Widget
   * @param config {Object} Configuration object (see Configuration attributes)
   * @constructor
   */
  function MonitorApp(config) {
    var t = this;
    t._rpc = config.rpc;
    MonitorApp.superclass.constructor.apply(this,arguments);
  }
	
  // Private module level static properties
  var getCN = Y.ClassNameManager.getClassName,
      CONTENT_BOX     = 'contentBox',
      BOUNDING_BOX    = 'boundingBox',
      DEFAULT_PAGE_ID = 'default-page';
      LOCAL_HOST_ID   = location.hostname + ':' + location.port,
      MONITOR_APP     = 'monitorapp',
      NAMESPACE       = 'Monitor',
      PROCESSES       = 'processes',
      PAGES           = 'pages',
      
      CLICK           = 'click',
      SELECTED        = 'selected',
      TAB_SELECTED    = 'tab-selected',
      DOT             = '.',
      POUND           = '#',
      TAB_BODY_PREFIX = 'body-',
      
      L               = Y.Lang,
      create          = Y.Node.create,
      isNumber        = L.isNumber,
      isString        = L.isString,
      merge           = Y.merge,
      substitute      = Y.substitute;

  // Extend with a YUI Widget, and this widget code
  var W = Y.MonitorApp = Y.extend(MonitorApp, Y.Widget, {

    /**
     * Category to prefix all event subscriptions to allow for ease of 
     * detach during destroy.
     *
     * @property _evtCat
     * @type string
     * @protected
     */
    _evtCat: null,

    /**
     * Persistent ID for this instance, used as the key to localStorage
     * for persistent attributes.  Set during construction based on the 
     * id attribute.
     *
     * @property _storageId
     * @type string
     * @protected
     */
    _storageId: null,

    /**
     * The object stored into local storage for this _storageId.
     *
     * @property _storage
     * @type object
     * @protected
     */
    _storage: null,

    /**
     * Reference to the Node instance containing the header contents.
     *
     * @property _header
     * @type Node
     * @protected
     */
    _header    : null,

    /**
     * Reference to the Node instance containing everything below 
     * the header.
     *
     * @property _body
     * @type Node
     * @protected
     */
    _body    : null,

    /**
     * Reference to the Node instance that will house the console messages.
     *
     * @property _browser
     * @type Node
     * @protected
     */
    _browser    : null,

    /**
     * Reference to the Node instance containing the desktop - the right
     * portion of the application that contains the tabs for switching
     * among monitor screens.
     *
     * @property _desktop
     * @type Node
     * @protected
     */
    _desktop    : null,

    /**
     * Reference to the Node instance that contains the page footer 
     *
     * @property _footer
     * @type Node
     * @protected
     */
    _footer    : null,

    /**
     * Reference to the JSON-RPC object for server interactions
     *
     * @property _rpc
     * @type {JSON_RPC}
     * @protected
     */
    _rpc : null,

    /**
     * Wrapper for <code>Y.log</code>.
     *
     * @method log
     * @param arg* {MIXED} (all arguments passed through to <code>Y.log</code>)
     * @chainable
     */
    log: function () {
      var t=this;
      Y.log.apply(Y,arguments);
      return t;
    },

    /**
     * Constructor code.  Initialize member variables from attributes
     * and browser localStorage.
     * 
     * @method initializer
     * @protected
     */
    initializer : function () {
      var t = this;
      t._evtCat = Y.stamp(t) + '|';
      t._storageId = W.NAME + '.' + t.get('id');

      // Initialize local storage persistence
      t._initLocalStorage();
    },
   
    /**
     * Tears down the instance, flushing event subscriptions and purging 
     * the UI.
     *
     * @method destructor
     * @protected
     */
    destructor : function () {
      var t = this;
      var bb = t.get(BOUNDING_BOX);
      bb.detach(t._evtCat + '*');
      bb.set('innerHTML','');
    },

    /**
     * Generate the user interface.  This method is called by the YUI 
     * framework once to render the DOM for the UI.
     *
     * @method renderUI
     * @protected
     */
    renderUI : function () {
      var t = this, cb = t.get(CONTENT_BOX);
      t._app = create(substitute(W.APP_TEMPLATE, W));
      cb.appendChild(t._app);
      t._header = cb.one(DOT + W.C_HEADER);
      t._body = cb.one(DOT + W.C_BODY);
      t._browser = cb.one(DOT + W.C_BROWSER);
      t._desktop = cb.one(DOT + W.C_DESKTOP);
      t._footer = cb.one(DOT + W.C_FOOTER);

      // Load each page and remote process
      for (var id in t._storage[PROCESSES]) {t._addProcess(id);}
      for (var id in t._storage[PAGES]) {t._addPage(id);}
t._addProcess("df7-8:4200");
    },

    /**
     * Sync the UI state to the current attribute state.
     *
     * @method syncUI
     */
    syncUI : function () {
      var t = this;
      /*
      t._uiUpdatePaused(t.get(PAUSED));
      t._uiSetHeight(t.get(HEIGHT));
      */
    },

    /**
     * Set up event listeners to wire up the UI to the internal state.
     *
     * @method bindUI
     * @protected
     */
    bindUI : function () {
      var t = this;
      // Setup click events
      var cb = t.get(CONTENT_BOX);
      cb.all(DOT + W.C_TABS + ' li').on(CLICK, t._onTabChange, t);

      // Attribute changes
      t.after(t._evtCat + 'stringsChange', t._afterStringsChange);
    },

    /**
     * Initialize from local storage, and add this connected host to
     * the list of processes if it's not there already.
     * 
     * @method _initLocalStorage()
     * @protected
     */
    _initLocalStorage : function () {
      var t = this;
      try {
        t._storage = JSON.parse(localStorage[t._storageId] || '{}');
      } catch (e) {
    	t._storage = {};
      }

      // Add some defaults
      if (!t._storage[PROCESSES]) {t._storage[PROCESSES] = {};}
      if (!t._storage[PAGES]) {t._storage[PAGES] = {};}
      if (!t._storage[PROCESSES][LOCAL_HOST_ID]) {
        t._storage[PROCESSES][LOCAL_HOST_ID] = {};
      }

      // Persist changes to local storage
      t._persist();
    },

    /**
     * Persist any configuration changes stored in the _storage object.
     * 
     * @method _persist()
     * @protected
     */
    _persist : function () {
      var t = this;
      localStorage[t._storageId] = JSON.stringify(t._storage);
    },

    /**
     * Add a new remote process widget to the browser
     *
     * @method _addProcess()
     * @param id {String} Remote process ID - must exist in localStorage
     * @protected
     */
    _addProcess : function (id) {

      var t = this;
      var cb = t.get(CONTENT_BOX);
      var params = {
        id:id,
        rpc:t._rpc
      };
      new Y.Monitor.RemoteProcess(params).render(cb.one(DOT + W.C_BROWSER));

    },

    /**
     * Add a desktop page tab
     *
     * @method _addPage()
     * @param id {String} Tab ID - must exist in localStorage
     * @protected
     */
    _addPage : function (id) {
    },

    /**
     * Process the desktop tab change event
     *
     * @method _onTabChange
     * @param e {Event} The event object - to determine the tab clicked
     * @protected
     */
    _onTabChange : function (e) {
      var t = this;
      var cb = t.get(CONTENT_BOX);
      var newTab = e.currentTarget;
      if (newTab.hasClass(TAB_SELECTED)) {return;}
      var newTabBody = cb.one(POUND + TAB_BODY_PREFIX + newTab.get('id'));
      cb.all(DOT + TAB_SELECTED).removeClass(TAB_SELECTED);
      newTab.addClass(TAB_SELECTED);
      newTabBody.addClass(TAB_SELECTED);
    },

    /**
     * Updates the UI if changes are made to any of the strings in the strings
     * attribute.
     *
     * @method _afterStringsChange
     * @param e {Event} Custom event for the attribute change
     * @protected
     */
    _afterStringsChange : function (e) {
        var prop   = e.subAttrName ? e.subAttrName.split(DOT)[1] : null,
            cb     = this.get(CONTENT_BOX),
            before = e.prevVal,
            after  = e.newVal;

        if ((!prop || prop === TITLE) && before.title !== after.title) {
            cb.all(DOT+C_CONSOLE_TITLE).set(INNER_HTML, after.title);
        }

        if ((!prop || prop === PAUSE) && before.pause !== after.pause) {
            cb.all(DOT+C_PAUSE_LABEL).set(INNER_HTML, after.pause);
        }

        if ((!prop || prop === CLEAR) && before.clear !== after.clear) {
            cb.all(DOT+C_CLEAR).set('value',after.clear);
        }
    }

  },

  // Y.MonitorApp public static properties
  {
    /**
     * The identity of the widget.
     *
     * @property MonitorApp.NAME
     * @type String
     * @static
     */
    NAME : MONITOR_APP,

    /**
     * HTML template for the application.
     *
     * @property MonitorApp.APP_TEMPLATE
     * @type String
     * @static
     */
    APP_TEMPLATE: [
      '<div class="{C_HEADER}">',
        'Node.js Monitor',
      '</div>',
      '<div class="yui3-g {C_BODY}">',
        '<div class="yui3-u {C_BROWSER}"></div>',
        '<div class="yui3-u {C_DESKTOP}">',
          '<ol class="{C_TABS}">',
            '<li id="tab-1" class="round-top">Tab 1</li>',
            '<li id="tab-2" class="round-top ' + TAB_SELECTED + '">Tab 2</li>',
            '<li id="tab-3" class="round-top">Tab 3</li>',
            '<li id="tab-4" class="round-top">+</li>',
          '</ol>',
          '<div class="{C_TABS_BODY}">',
            '<div id="' + TAB_BODY_PREFIX + 'tab-1" class="{C_TAB_BODY}">Tab body 1</div>',
            '<div id="' + TAB_BODY_PREFIX + 'tab-2" class="{C_TAB_BODY} ' + TAB_SELECTED + '">Tab body 2</div>',
            '<div id="' + TAB_BODY_PREFIX + 'tab-3" class="{C_TAB_BODY}">Tab body 3</div>',
            '<div id="' + TAB_BODY_PREFIX + 'tab-4" class="{C_TAB_BODY}">Tab body 4</div>',
          '</div>',
        '</div>',
      '</div>',
      '<div class="{C_FOOTER}">',
        'Released on <a href="https://github.com/lorenwest/node-monitor">github</a> under the <a href="https://github.com/lorenwest/node-monitor/blob/master/LICENSE">Apache License 2.0</a>',
        '<span class="subtitle">version 0.3.0</span>',
      '</div>',
    ''].join('\n'),

    C_HEADER        : getCN(MONITOR_APP, 'header'),
    C_BODY          : getCN(MONITOR_APP, 'body'),
    C_BROWSER       : getCN(MONITOR_APP, 'browser'),
    C_DESKTOP       : getCN(MONITOR_APP, 'desktop'),
    C_TABS          : getCN(MONITOR_APP, 'tabs'),
    C_TABS_BODY     : getCN(MONITOR_APP, 'tabs-body'),
    C_TAB_BODY      : getCN(MONITOR_APP, 'tab-body'),
    C_FOOTER        : getCN(MONITOR_APP, 'footer')

  });

  /* Add this class to my application namespace */
  Y.namespace(NAMESPACE).MonitorApp = MonitorApp;

}, "0.3.0", {requires:["widget", "remoteprocess", "yui-log", "substitute"]});
