var sys = require('sys'),
    events = require('events'),
    Config = require('config'),
    Monitor = require('./monitor'),
    MonitorEvent = require('./monitor-event');

/**
 * <p>Base class from which probes are derived.</p>
 *
 * <p>A probe watches for interesting things, and emits monitor events when it
 * has something to report.
 * </p>
 *
 * @class Probe
 * @extends events.EventEmitter
 * @constructor
 * @param {String} classId - The registered class ID
 * @param {String} name - A name identifying the instance of this probe.
 * @param {Object} config - Public instance configurations
 */
var Probe = module.exports = function (classId, name, config) {

    var t = this, classDef = Monitor.getClassDefinition(classId);
    events.EventEmitter.call(this);
    t.classId = classId;
    t.name = name ? name : classDef.name;
    t.config = Config._extendDeep({}, classDef.defaultParameters, config);
    t.isRunning = false;

    // Register this probe instance
    Monitor.registerProbe(t);
}
sys.inherits(Probe, events.EventEmitter);

/**
 * <p>Start the probe.</p>
 *
 * <p>
 *   Sub-classes that override start/stop in order to obtain/release resources must
 *   assure this method is called.
 * </p>
 *
 * @method start
 * @param {Function} callback(error) - (optional) Called when the probe has started
 */
/**
 * The probe has started.
 * @event start
 * @param {Object} event - event.probeId: ProbeId that has started.
 */
Probe.prototype.start = function (callback) {
  var t = this;
  t.isRunning = true;
  t.emit('start', {probeId:t.id});
  if (callback) callback(null);
}

/**
 * <p>Stop the probe.</p>
 *
 * <p>
 *   Sub-classes that override start/stop in order to obtain/release resources must
 *   assure this method is called.
 * </p>
 *
 * @method stop
 * @param {Function} callback(error) - (optional) Called when the probe has stopped
 */
/**
 * The probe has stopped.
 * @event stop
 * @param {Object} event - event.probeId: ProbeId that has stopped.
 */
Probe.prototype.stop = function (callback) {
  var t = this;
  t.isRunning = false;
  t.emit('stop', {probeId:t.id});
  if (callback) callback(null);
}

/**
 * Emit a monitor event.
 *
 * This method is called by sub-classes to emit events being monitored.
 * @method emitMonitorEvent
 * @param {String} eventName - Name of the event being monitored
 * @param {Object} data - A data object containing additional information about the event.
 */
/**
 * This is the event fired by the probe when it has something to monitor
 * @event monitor
 * @param {MonitorEvent} event - The object representing the event to be monitored
 */
Probe.prototype.emitMonitorEvent = function(eventName, data) {
  var t = this;
  t.emit('monitor', new MonitorEvent(eventName, t.id, data));
}
