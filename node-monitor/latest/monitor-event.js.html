/**
 * Monitor event model
 *
 * These objects are created by probes, and sent when they emit the monitor
 * event.  They represent a single event that happened in time.
 *
 * These objects pass through filters and between processes.
 *
 * @class MonitorEvent
 * @constructor
 * @param {String} name - Name of the event being monitored
 * @param {Integer} pId - The registered ID of the probe that generated the event
 * @param {Object} data - Additional data to monitor (optional)
 */
var MonitorEvent = function (name, pId, data) {
    this.stamp = new Date();
    this.name = name;
    this.pId = pId;
    this.data = data;
}
