// Dependencies
var sys = require('sys'),
    events = require('events'),
    Config = require('config'),
    Monitor = require('./monitor');

/**
 * <p>Base class from which all monitors are derived.</p>
 *
 * @class MonitorBase
 * @extends events.EventEmitter
 * @constructor
 * @param {String} classId - The registered class ID
 * @param {String} name - A name identifying the instance of this monitor.
 * @param {Object} config - Public instance configurations
 */
var MonitorBase = module.exports = function (classId, name, config) {

    var t = this, classDef = Monitor.getClassDefinition(classId);
    events.EventEmitter.call(this);
    t.classId = classId;
    t.name = name ? name : classDef.name;
    t.config = Config._extendDeep({}, classDef.defaultParameters, config);
    t.isMonitoring = false;

    // Register this monitor instance
    t.monitorId = Monitor.registerMonitor(t);
}
sys.inherits(MonitorBase, events.EventEmitter);

/**
 * <p>Start monitoring.</p>
 *
 * <p>
 *   Sub-classes that override start/stop in order to obtain/release resources must
 *   assure this method is called.
 * </p>
 *
 * @method start
 */
/**
 * The monitor has started.
 * @event start
 * @param {Object} event - event.mId: MonitorID that has started.
 */
MonitorBase.prototype.start = function () {
  var t = this;
  t.isMonitoring = true;
  t.emit('start', {mId:t.monitorId});
}

/**
 * <p>Stop monitoring.</p>
 *
 * <p>
 *   Sub-classes that override start/stop in order to obtain/release resources must
 *   assure this method is called.
 * </p>
 *
 * @method stop
 */
/**
 * The monitor has stopped.
 * @event stop
 * @param {Object} event - event.mId: MonitorID that has stopped.
 */
MonitorBase.prototype.stop = function () {
  var t = this;
  t.isMonitoring = false;
  t.emit('stop', {mId:t.monitorId});
}

/**
 * Emit a monitor event.
 *
 * This method is called by sub-classes to emit events being monitored.
 * @method monitorEvent
 * @param {String} eventName - Name of the event being monitored
 * @param {Object} data - A data object containing additional information about the event.
 */
/**
 * This is the main event fired whenever there's something to monitor.
 * @event monitor
 * @param {MonitorEvent} event - The object representing the monitor event
 */
MonitorBase.prototype.monitorEvent = function(eventName, data) {
  var t = this;
  t.emit('monitor', new MonitorEvent(eventName, t.monitorId, data));
}

/**
 * Monitor event model
 *
 * Instances of this class represent a single monitored event.
 *
 * @class MonitorEvent
 * @constructor
 * @param {String} name - Name of the event being monitored
 * @param {Integer} mId - The registered ID of the monitor that generated the event
 * @param {Object} data - Additional data to monitor (optional)
 */
var MonitorEvent = function (name, mId, data) {
    this.stamp = new Date();
    this.name = name;
    this.mId = mId;
    this.data = data;
}
