// Dependencies
var sys = require('sys'),
    events = require('events'),
    Monitor = require('./monitor'),
    MonitorBase = require('./monitor-base');

// Define and register the monitor class
var classDefinition = {
  name: 'Process Monitor',
  description: 'Monitors all process related information and events.  Static process information (pid, etc) is located in the monitor preferences.  Monitor events are generated on a polling interval.',
  defaultParameters: {
    pollInterval: '0 * * * * *'
  },
  parameterDescriptions: {
    pollInterval: 'Polling interval for generating monitor events. The cron pattern consists of six fields separated by spaces. The format is: [second] [minute] [hour] [day of month] [month] [day of week].  More about cron formats here: http://help.sap.com/saphelp_xmii120/helpdata/en/44/89a17188cc6fb5e10000000a155369/content.htm'
  }
};

/**
 * Monitor the running node.js process.
 *
 * @class ProcessMonitor
 * @extends MonitorBase
 * @constructor
 * @param {String} name - A name identifying the instance of this monitor.
 * @param {Object} config - Configuration override object
 */
var ProcessMonitor = module.exports = function (name, config) {
  MonitorBase.call(this, classId, name, config);
}
sys.inherits(ProcessMonitor, MonitorBase);
var classId = Monitor.registerClass(ProcessMonitor, classDefinition);

/**
 * Start the monitor
 *
 * @method start
 */
ProcessMonitor.prototype.superStart = ProcessMonitor.prototype.start;
ProcessMonitor.prototype.start = function() {
  var t = this;
  t.superStart();
  console.error("Starting the process monitor");
}
